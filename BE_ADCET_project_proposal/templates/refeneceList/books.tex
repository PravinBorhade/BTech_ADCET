\noindent{\Large{\textbf{6. Reference List}}}
~\\~\\~
Use standard IEEE Format
~\\~\\
\noindent{\large{\textbf{Articles from Journals}}}
~\\~\\~
\noindent{\large{\textbf{Basic Format:  }}}
~\\~\\~
J. K. Author, “Name of paper,” Abbrev. Title of Periodical, vol. x, no. x, pp. xxx-xxx, Abbrev. Month, year.
~\\~\\~
J. K. Author, “Name of paper,” Abbrev. Title of Periodical, vol. x, no. x, pp. xxx-xxx, Abbrev. Month, year, doi: xxx. 
~\\~\\~
\noindent{\large{\textbf{Examples:  }}}
~\\~\\~
\noindent{[1] S. G. Sapate and S. N. Talbar, “Radiomics based detection and characterization of suspicious lesions on full field digital mammograms,” Computer methods and programs in biomedicine, vol. 163, pp. 1–20, May. 2018, doi: 10.1016/j.cmpb.2018.05.017.}
~\\~\\~
\noindent{[2] S. G. Sapate et al., “Breast cancer diagnosis using abnormalities on ipsilateral views of digital mammograms,” Biocybernetics and Biomedical Engineering, vol. 40, no. 1, pp. 290–305, Jan. 2020.}
~\\~\\~
\noindent{\large{\textbf{Books }}}
~\\~\\~
\noindent{\large{\textbf{Book Basic Format: }}}
~\\~\\~
\noindent{J. K. Author, “Title of chapter in the book,” in Title of His Published Book, xth ed. City of Publisher, (only. U.S. State), Country: Abbrev. of Publisher, year, ch. x, sec. x, pp. xxx–xxx. Examples: II.}
~\\~\\~
\noindent{Klaus and P. Horn, Robot Vision. Cambridge, MA, USA: MIT Press, 1986.}
~\\~\\~
\noindent{L. Stein, “Random patterns,” in Computers and You, J. S. Brake, Ed. New York, NY, USA: Wiley, 1994, pp. 55–70.}
~\\~\\~
\noindent{\large{\textbf{Book With Chapter Title}}}
~\\~\\~
\noindent{\large{\textbf{Basic Format:  }}}
~\\~\\~
J. K. Author, “Title of chapter in the book,” in Title of Published Book, X. Editor, Ed., City of Publisher, State (only U.S.), Country: Abbrev. of Publisher, year, ch. x, sec. x, pp. xxx–xxx. 
~\\~\\~
Examples:
~\\~\\~
\noindent{T. Ogura, “Electronic government and surveillance-oriented society,” in Theorizing Surveillance: The Panopticon and Beyond. Cullompton, U.K.: Willan, 2006, ch. 13, pp. 270–295.}
~\\~\\~
\noindent{L. Li, J. Yang, and C. Li, “Super-resolution restoration and image reconstruction for passive millimeter wave imaging,” in Image Restoration—Recent Advances and Applications, A. Histace, Ed., Rijeka, Croatia: InTech, 2012, pp. 25–45.}
